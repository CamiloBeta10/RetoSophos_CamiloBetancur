function GetSearchMadeUser()
{
    var bnwsc = '';
    ca = decodeURIComponent(document.cookie).split(';');
    for (var bnwi = 0; bnwi < ca.length; bnwi++) {
        if (ca[bnwi].indexOf("VTEXSC") > -1) {
            bnwsc = ca[bnwi].split('=')[2];
        }
    }
    if (bnwsc=='') bnwsc=15;
    $.ajax({
        url: 'https://u.braindw.com/api/VtexApi/GetSearchMade/',
        dataType: 'json',
        cache: false,
        async: true,
        xhrFields: {
            withCredentials: true
        },
        crossDomain: true,
        timeout: 20000,
        data: 'token=<%TOKEN%>&company=metrofood&url=' + window.location.host + '&BWGuid=&branchOffice=' + bnwsc,
        success: function (data) {
            BraindwRendeSearch(data);

}

function parseJwt (token) {
    try{
    var base64Url = token.split('.')[1];
    var base64 = base64Url.replace('-', '+').replace('_', '/');
    return JSON.parse(window.atob(base64));
    }catch(e)
    { return null;}
}

function clearProductsBwnGond(IdCode,sufix) {
    try {
        var dataAdic = " kg";
        if ($('#brandwProductGond-' + sufix + '-' + IdCode).val().indexOf(',') > -1 || $('#brandwProductGond-' + sufix + '-' + IdCode).val().indexOf('.') > -1) {
            dataAdic = " kg";
        } else {
            dataAdic = "";
        }
        var actual = parseFloat($('input[id=cantProdBwnGond-' + sufix + '-' + IdCode +']').val().replace(' kg', ''));
        var peso = parseFloat($('#brandwProductGond-' + sufix + '-' + IdCode).val().replace(' kg', ''));
        if ((actual - peso) <= 0) {
            $('input[id=cantProdBwnGond-' + sufix + '-' + IdCode +']').val(peso + dataAdic); 
            $('#clearProductsBwnGond-' + sufix + '-' + IdCode).removeClass('brandwdecrease');
            $('#clearProductsBwnGond-' + sufix + '-' + IdCode).addClass('brandwdecreasedisable');
        } else {
            $('#clearProductsBwnGond-' + sufix + '-' + IdCode).removeClass('brandwdecreasedisable');
            $('#clearProductsBwnGond-' + sufix + '-' + IdCode).addClass('brandwdecrease');
            if (dataAdic == "") {
                $('input[id=cantProdBwnGond-' + sufix + '-' + IdCode +']').val((actual - peso));
            } else {
                var pesoFinal = (actual - peso).toFixed(2);
                if (pesoFinal[pesoFinal.length - 1] == '0')
                    pesoFinal = pesoFinal.substr(0, pesoFinal.length - 1);

                $('input[id=cantProdBwnGond-' + sufix + '-' + IdCode +']').val(pesoFinal + dataAdic);
            }
        }
    } catch (e) {}
}

    var cantitems = 6;
    var sliderItemWidth = 'width:14.8%;';
    var sliderConfigStyle='width: 100%; display: block; opacity: 1; margin-right:auto; margin-left:2.2%';
    var titleConfigured = '';
    var displayTitulo ='none';
    var navigationContainerWidth='';
    var styleContent= '';

    if (title != "") {
        displayTitulo ='block';
    }

    if(configs != undefined && configs != null && configs != "" && bdwMobile.any() == null){
        styleContent= configs.Content;
        cantitems = configs.carouselItemsNum;
        sliderItemWidth = configs.sliderItemWidth;
        sliderConfigStyle = configs.sliderConfigStyle;

        for(var i = 0; i < configs.imagesObjects.length ; i++){
            var img = configs.imagesObjects[i];
            divsImagesAppend[i] = BdwBuildImageObject(img.url, img.linkImg, img.imageWidth, img.marginSide, img.floatImage,img.paddingImage,img.borderRadiusImage, img.styleContent);
            divsImagesAppend[i] = divsImagesAppend[i].replace(/%eventoimg%/g,' bw-title="' + parentId +'" bw-sku="" bw-section="' + eventName +'"' );
            if (linkViewMore=='')
              linkViewMore = img.linkImg;
        }
        navigationContainerWidth=configs.navigationContainerWidth;
        if (configs.title.value==''|| configs.title.value==null)
          displayTitulo = 'none';
        else
            title=configs.title.value;
    }

    if (bdwMobile.any() != null) {
        cantitems = 2;
        sliderItemWidth = 'width:45.8%;';
        styleContent='';
    }

    
    titleConfigured = '<div class="header clear-fix" style="display:' + displayTitulo + ';">'+
       '<h2 class="title"><span class="text braindwtxt23">'+title+'</span></h2>'+
       '</div>';

    if (linkViewMore=='' && linkImgBdw!='')
    {
        linkViewMore=linkImgBdw;
    }

    
    if(selector != 'braindw_history_dinamic'){
        var resto = data.length % cantitems;
        data.splice(data.length - resto, resto);
    } 
    if (data.length > 0) {
